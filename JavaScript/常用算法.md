# js常用算法

## 1.排序

1）冒泡排序：

```
var arr = [10, 20, 5, 8, 21, 100, 99];
        function test() {
            for(var i = 0; i < arr.length; i++) {
                for(var j= 0; j < arr.length-i-1; j++) {
                    if(arr[j] > arr[j+1]) {
                        var temp = arr[j+1];
                        arr[j+1] = arr[j];
                        arr[j] = temp;
                    }
                }
            }
            document.write(arr);
        }
test();
```

2）sort排序：

```
var a = [10, 20, 5, 8, 21, 100, 99];
        a.sort(function(n1,n2){
            return n1-n2; //    n1-n2：升序  n2-n1：降序
        });
        document.write(a);
```
