# js常用算法

## 1.排序

1）冒泡排序：

```
var arr = [10, 20, 5, 8, 21, 100, 99];
        function test() {
            for(var i = 0; i < arr.length; i++) {
                for(var j= 0; j < arr.length-i-1; j++) {
                    if(arr[j] > arr[j+1]) {
                        var temp = arr[j+1];
                        arr[j+1] = arr[j];
                        arr[j] = temp;
                    }
                }
            }
            document.write(arr);
        }
test();
```

2）sort排序：

```
var a = [10, 20, 5, 8, 21, 100, 99];
        a.sort(function(n1,n2){
            return n1-n2; //    n1-n2：升序  n2-n1：降序
        });
        document.write(a);
```

## 2.去重：

1）数组去重：

```
var arr = [1,2,2,3,4,4,5];
arr.sort(function(n1,n2){
            return n1-n2
        });
        for(i=0;i<arr.length;i++){
            for(j=i;j<arr.length-1;j++){
                if(arr[i] == arr[j+1]){
                    arr.splice(j+1,1)；
                    j--:
                }
            }
        }
console.log(arr);
//ES6
var arr1 = [...new  Set(arr)];
console.log(arr1);
```

2）字符串去重：

```
 var str = 'abbcccddde';
        var str1 = str.split('');
        for(i=0;i<str1.length;i++){
            for(j=i;j<str1.length-1;j++){
                if(str1[i] == str1[j+1]){
                    str1.splice(j+1,1)
                    j--;
                }
            }
        }
console.log(str1.join(''));
//ES6
var str2 = [...new Set(str)].join('');
console.log(str2);
```

## 3.两个变量交换顺序（不借助第三个变量）：

```
var a = 10;
var b = 20;
a = a + b;
b = a - b;
a = a - b;
console.log(a);
console.log(b);
//ES6
let a = 10;
let b = 20;
[a,b] = [b,a]

```
